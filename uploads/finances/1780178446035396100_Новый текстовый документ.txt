CREATE TABLE admin.user_tab (
  user_id SERIAL,
  fio VARCHAR NOT NULL,
  email VARCHAR NOT NULL,
  password VARCHAR NOT NULL,
  user_role_id INTEGER NOT NULL,
  phone_num VARCHAR NOT NULL,
  rowversion TIMESTAMP WITHOUT TIME ZONE DEFAULT now() NOT NULL,
  CONSTRAINT user_tab_pkey PRIMARY KEY(user_id),
  CONSTRAINT user_tab_fk FOREIGN KEY (user_role_id)
    REFERENCES admin.user_role_tab(user_role_id)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
    NOT DEFERRABLE
) 
WITH (oids = false);

ALTER TABLE admin.user_tab
  OWNER TO postgres;
